  var child, i, j, len, len1, page, pages, ref1, ref2;
  pages = [];

    pages.push(section);

};
 
findChapterSection = function(section, chapter_section) {
  var child, found, i, len, ref1;
  if (_.isEqual(section.chapter_section, chapter_section)) {

};
